Smiley Separation Model

This is an improvement to our smiley separation which now takes the 3 turns as separate inputs into the LSTM layers.

Using TensorFlow backend.

Config

This model was produced with the following configs:

Embedding Dim = 200
Batch Size = 200
LSTM Dim = 128
Dropout = 0
Learning Rate = 0
Num Epochs = 100
Early Stopping = True

Model Specification

An extra hidden layer which is supposed to extract the important information from the

first embedding then add information from smilys

def buildModel(embeddingMatrix, smileyEmbeddings):
    """Constructs the architecture of the model
    Input:
        embeddingMatrix : The embedding matrix to be loaded in the embedding layer.
    Output:
        model : three layer lstm model with one layer of meta data
    """

    x1 = Input(shape=(100,), dtype='int32', name='main_input1')
    x2 = Input(shape=(100,), dtype='int32', name='main_input2')
    x3 = Input(shape=(100,), dtype='int32', name='main_input3')
    smiley_input = Input(shape=(20,), dtype='int32', name='main_input4')

    #pretrained embedding layers
    embeddingLayer = Embedding(embeddingMatrix.shape[0],
                                EMBEDDING_DIM,
                                weights=[embeddingMatrix],
                                input_length=MAX_SEQUENCE_LENGTH,
                                trainable=False)

    smileyEmbeddingLayer = Embedding(smileyEmbeddings.shape[0],
                                300,
                                weights=[smileyEmbeddings],
                                input_length=20,
                                trainable=False)

    emb1 = embeddingLayer(x1)
    emb2 = embeddingLayer(x2)
    emb3 = embeddingLayer(x3)

    #smiley embeddings
    emb_smiley = smileyEmbeddingLayer(smiley_input)

    #LSTM layers, need to define a new one for different embeddings
    lstm = LSTM(LSTM_DIM, dropout=DROPOUT)
    lstm_smiley = LSTM(LSTM_DIM, dropout=0.2)

    lstm1 = lstm(emb1)
    lstm2 = lstm(emb2)
    lstm3 = lstm(emb3)

    lstm_smiley = lstm_smiley(emb_smiley)

    #full network
    concatenated_out = Concatenate(axis=-1)([lstm1, lstm2, lstm3])

    #cool hidden
    hidden_layer = Dense(512, activation='relu')(concatenated_out)
    concatenate_emoji = Concatenate(axis=-1)([hidden_layer, lstm_smiley])
    dropout = Dropout(0.2)(concatenate_emoji)

    #output
    model_output = Dense(4, activation='sigmoid')(dropout)
    model = Model([x1, x2, x3, smiley_input], model_output)

    rmsprop = optimizers.rmsprop(lr=LEARNING_RATE)
    adam =  optimizers.adam(lr=LEARNING_RATE)

    model.compile(loss='categorical_crossentropy',
                  optimizer=adam,
                  metrics=['acc'])

    return model

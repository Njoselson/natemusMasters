Model Specification

An extra hidden layer which is supposed to extract the important information from the

first embedding then add information from smilys

def buildModel(embeddingMatrix, smileyEmbeddings):
    """Constructs the architecture of the model
    Input:
        embeddingMatrix : The embedding matrix to be loaded in the embedding layer.
    Output:
        model : three layer lstm model with one layer of meta data
    """

    x1 = Input(shape=(100,), dtype='int32', name='main_input1')
    x2 = Input(shape=(100,), dtype='int32', name='main_input2')
    x3 = Input(shape=(100,), dtype='int32', name='main_input3')
    smiley_input = Input(shape=(20,), dtype='int32', name='main_input4')

    #pretrained embedding layers
    embeddingLayer = Embedding(embeddingMatrix.shape[0],
                                EMBEDDING_DIM,
                                weights=[embeddingMatrix],
                                input_length=MAX_SEQUENCE_LENGTH,
                                trainable=True)

    smileyEmbeddingLayer = Embedding(smileyEmbeddings.shape[0], 300, weights=[smileyEmbeddings], input_length=20, trainable=True)

    emb1 = embeddingLayer(x1)
    emb2 = embeddingLayer(x2)
    emb3 = embeddingLayer(x3)

    emb1 = Dropout(DROPOUT)(emb1)
    emb2 = Dropout(DROPOUT)(emb2)
    emb3 = Dropout(DROPOUT)(emb3)


    #smiley embeddings
    emb_smiley = smileyEmbeddingLayer(smiley_input)
    emb_smiley = Dropout(DROPOUT)(emb_smiley)

    #LSTM layers, need to define a new one for different embeddings
    lstm_person1 = Bidirectional(GRU(LSTM_DIM, dropout=DROPOUT,recurrent_dropout=0.5, return_sequences=True))
    lstm_person2 = Bidirectional(GRU(LSTM_DIM, dropout=DROPOUT,recurrent_dropout=0.5, return_sequences=True))
    lstm_smiley = Bidirectional(GRU(int(LSTM_DIM/2), dropout=DROPOUT,recurrent_dropout=0.5, return_sequences=False))

    lstm_final= Bidirectional(GRU(LSTM_DIM, dropout=DROPOUT,recurrent_dropout=0.5, return_sequences=False))

    lstm1 = lstm_person1(emb1)
    lstm2 = lstm_person2(emb2)
    lstm3 = lstm_person1(emb3)

    lstm4 = lstm_smiley(emb_smiley)

    #full network
    concatenated_lstm = Concatenate(axis=-1)([lstm1, lstm2, lstm3])
    reshaped_lstm = Flatten()(concatenated_lstm)

    concatenated_smiley = Concatenate(axis=-1)([reshaped_lstm, lstm4])

    #cool hidden
    hidden_layer = Dense(256, activation='relu')(concatenated_smiley)
    dropout = Dropout(DROPOUT)(hidden_layer)
    #final_lstm = lstm_final(concatenated_smiley)
    #dropout = Dropout(DROPOUT)(final_lstm)

    #output
    model_output = Dense(4, activation='softmax')(dropout)
    model = Model([x1, x2, x3, smiley_input], model_output)

    rmsprop = optimizers.rmsprop(lr=LEARNING_RATE)
    adam =  optimizers.adam(lr=LEARNING_RATE)

    model.compile(loss='categorical_crossentropy',
                  optimizer=rmsprop,
                  metrics=['acc'])

    return model
